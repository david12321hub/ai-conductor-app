import stripe

stripe.api_key = os.getenv("STRIPE_SECRET_KEY")

# ==================== Payment Tiers & Pay-as-you-go ====================
def show_plans():
    st.header("Choose Your Plan")
    st.caption("Unlimited access or top up credits anytime")

    col1, col2, col3 = st.columns(3)

    # Pro Unlimited
    with col1:
        st.subheader("Pro Unlimited")
        st.write("$49/month")
        st.caption("Unlimited light use + 10K tokens/day")
        if st.button("Choose Pro Unlimited", key="pro_unlimited"):
            try:
                session = stripe.checkout.Session.create(
                    payment_method_types=['card'],
                    line_items=[{'price': 'price_1TCRYTFC68YihsMHeQmcuabi', 'quantity': 1}],
                    mode='subscription',
                    success_url='https://aiconductorapp.com/success',
                    cancel_url='https://aiconductorapp.com/cancel',
                    client_reference_id=st.session_state.user.id if st.session_state.user else None,
                )
                st.markdown(f"[Pay Now →]({session.url})")
            except Exception as e:
                st.error(f"Payment error: {str(e)}")

    # Enterprise Unlimited
    with col2:
        st.subheader("Enterprise Unlimited")
        st.write("$149/month")
        st.caption("50K tokens/day + priority support")
        if st.button("Choose Enterprise", key="enterprise_unlimited"):
            try:
                session = stripe.checkout.Session.create(
                    payment_method_types=['card'],
                    line_items=[{'price': 'price_1TCRVKFC68YihsMHX5dYRXcJ', 'quantity': 1}],
                    mode='subscription',
                    success_url='https://aiconductorapp.com/success',
                    cancel_url='https://aiconductorapp.com/cancel',
                )
                st.markdown(f"[Pay Now →]({session.url})")
            except Exception as e:
                st.error(f"Payment error: {str(e)}")

    # Pay-as-you-go (one-time credit packs)
    with col3:
        st.subheader("Pay-as-you-go")
        st.write("Top up credits")
        amounts = [5, 10, 15, 20]
        selected_amount = st.selectbox("Select amount ($)", amounts, key="payg_amount")

        if st.button(f"Buy ${selected_amount} Credits", key="payg_buy"):
            # Map selected amount to the correct Price ID
            price_ids = {
                5: 'price_1TCRWnFC68YihsMHNsgK5Uld',
                10: 'price_1TCRYTFC68YihsMHeQmcuabi',
                15: 'price_1TCRZ6FC68YihsMHStTerTSD',
                20: 'price_1TCRaNFC68YihsMH19Smqd3P'
            }
            selected_price = price_ids.get(selected_amount)

            if selected_price:
                try:
                    session = stripe.checkout.Session.create(
                        payment_method_types=['card'],
                        line_items=[{'price': selected_price, 'quantity': 1}],
                        mode='payment',
                        success_url='https://aiconductorapp.com/success?amount=' + str(selected_amount),
                        cancel_url='https://aiconductorapp.com/cancel',
                        client_reference_id=st.session_state.user.id if st.session_state.user else None,
                    )
                    st.markdown(f"[Pay Now →]({session.url})")
                except Exception as e:
                    st.error(f"Payment error: {str(e)}")
            else:
                st.error("Price not found for selected amount.")
