def get_daily_tokens_used():
    if not st.session_state.user:
        return 0
    user_id = st.session_state.user.id
    today = datetime.date.today().isoformat()
    result = supabase.table("user_usage").select("tokens_used").eq("user_id", user_id).eq("date", today).execute()
    return result.data[0]["tokens_used"] if result.data else 0

def add_tokens_used(tokens):
    user_id = st.session_state.user.id
    today = datetime.date.today().isoformat()
    current = get_daily_tokens_used()
    supabase.table("user_usage").upsert({
        "user_id": user_id,
        "date": today,
        "tokens_used": current + tokens
    }, on_conflict="user_id, date").execute()

# Before every AI call (add this check)
def check_token_cap():
    tier = get_user_tier()
    cap = DAILY_TOKEN_CAP.get(tier, 5000)
    used = get_daily_tokens_used()
    if used >= cap:
        st.warning(f"Daily token limit reached for your tier ({tier}). Upgrade or wait until tomorrow.")
        st.stop()
    return True
